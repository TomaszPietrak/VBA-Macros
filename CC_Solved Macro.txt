Private Sub RemoveFillF2PayFeedbackTable()

    Dim ws As Worksheet
        Set ws = ThisWorkbook.Worksheets("F2 PAY Feedback")
        ws.Columns(1).Interior.ColorIndex = 0
    
End Sub

    
Private Sub F2PayDrillDown2Levels()
    
        Dim ws As Worksheet
        Dim index As Integer
        Set ws = ThisWorkbook.Worksheets("F2 PAY")
    
        Dim pt As PivotTable
        Set pt = ws.PivotTables("F2 Pay Pivot")
        
        Dim rngData As Range
        Set rngData = pt.DataBodyRange
        
        Dim c As Range
        For Each c In rngData.Rows
            If c.PivotCell.RowItems.Count = pt.RowFields.Count Then
                If c.PivotCell.RowItems(1).Name = "F2 Contact Center" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 EDM" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 Benefits" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                
                ElseIf c.PivotCell.RowItems(1).Name = "F2 Mobility" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 EE Relations" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 Expenses" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 Learning" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                ElseIf c.PivotCell.RowItems(1).Name = "F2 Perf Management" And c.PivotCell.RowItems(2).Name = "Pay" Then
                        c.ShowDetail = True
                    Call F2PayCopyData
                    Call F2PayDeletePageMacro
                    
                End If
            End If
        Next c
    End Sub
    
    
Private Sub F2PayCopyData()
    On Error GoTo ErrorHandler
    
    Dim ws As Worksheet
        Set ws = ThisWorkbook.Worksheets("F2 Pay")
    Dim index As Integer
    Dim WS2 As Worksheet
        index = ws.index - 1
        Set WS2 = ThisWorkbook.Worksheets(index)
        
    If WS2.Name = "Timeliness Feedback" Then
        GoTo ErrorHandler
    End If
    
    Dim myrng As Range
    Dim LastRow As Long
    LastRow = Cells.Find(What:="*", _
                    After:=Range("A1"), _
                    LookAt:=xlPart, _
                    LookIn:=xlFormulas, _
                    SearchOrder:=xlByRows, _
                    SearchDirection:=xlPrevious, _
                    MatchCase:=False).Row
    
    Dim LastRow2 As Long
    LastRow2 = Sheets("F2 Pay Feedback").Cells(Rows.Count, "A").End(xlUp).Row
    
    Dim LastOldDataRow As Long
    Dim rngNewRange As Range

    LastOldDataRow = LR(Sheets("F2 Pay Feedback"), 1)
    
    WS2.Range("C2", "C" & LastRow).Copy
    Worksheets("F2 PAY Feedback").Paste Destination:=Sheets("F2 PAY Feedback").Range("A" & LastRow2).Offset(1, 0)
        Set rngNewRange = Sheets("F2 Pay Feedback").Range(Sheets("F2 Pay Feedback").Cells(LastOldDataRow + 1, 1), Sheets("F2 Pay Feedback").Cells(LR(Sheets("F2 Pay Feedback"), 1), 1))
        rngNewRange.Interior.Color = vbYellow
    
    WS2.Range("D2", "D" & LastRow).Copy
    Worksheets("F2 PAY Feedback").Paste Destination:=Sheets("F2 PAY Feedback").Range("B" & LastRow2).Offset(1, 0)

    WS2.Range("S2", "S" & LastRow).Copy
    Worksheets("F2 PAY Feedback").Paste Destination:=Sheets("F2 PAY Feedback").Range("C" & LastRow2).Offset(1, 0)

    WS2.Range("T2", "T" & LastRow).Copy
    Worksheets("F2 PAY Feedback").Paste Destination:=Sheets("F2 PAY Feedback").Range("D" & LastRow2).Offset(1, 0)

    WS2.Range("BZ2", "BZ" & LastRow).Copy
    Worksheets("F2 PAY Feedback").Paste Destination:=Sheets("F2 PAY Feedback").Range("E" & LastRow2).Offset(1, 0)
    
    Exit Sub

ErrorHandler:
    ErrorOccurred = True
    
    End Sub
    

Private Function LR(xSh As Worksheet, xCol As Long) As Long 'last row
    With xSh
        LR = .Cells(.Rows.Count, xCol).End(xlUp).Row
    End With
End Function

    
Private Sub F2PayDeletePageMacro()
    On Error GoTo ErrorHandler

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets("F2 PAY")
    Dim index As Integer
    Dim WS2 As Worksheet
    index = ws.index - 1
     Set WS2 = ThisWorkbook.Worksheets(index)
    Debug.Print WS2.Name
    
'    If WS2.Name <> Left("Sheet", 5) Then
'    ErrorOccurred = True
'    GoTo ErrorHandler

    If WS2.Name = "Timeliness Feedback" Then

    WS2.Delete

    End Sub
    
    
Private Sub F2PayFinalMacro()

    End Sub
